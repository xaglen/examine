  333. /*****
  334.  * processPhone()
  335.  * formats a phone number for consistent entry into database
  336.  *****/
  337. function processPhone($element) {
  338.     // strips everything that isn't a digit
  339.     $element = preg_replace("/\D/", "", $element);
  340.     // convert to array of digits
  341.     $d = preg_split("//", $element);
  342.     
  343.     // remove empty cells at beginning and end
  344.     array_shift($d);
  345.     array_pop($d);
  346.     $digits = count($d);
  347.     if ( $digits < 10 ) return (false);    // must have at least 10 digits
  348.     // gives standard U.S. formatting of "123-123-1234"
  349.     $phoneNumber = $d[0].$d[1].$d[2]."-".$d[3].$d[4].$d[5]."-".$d[6].$d[7].$d[8].$d[9];
  350.     // allows for more digits (extensions, etc.) "123-123-1234 1234"
  351.     if ( $digits > 10 ) {
  352.         $phoneNumber .= " ";
  353.         for ( $i=10; $i<$digits; $i++ ) {
  354.             $phoneNumber .= $d[$i];
  355.         }
  356.     }
  357.     return($phoneNumber);
  358. }
