  369. function modifyUser () {
  370.     $uid = getOne("SELECT uid FROM users WHERE userName='".$_POST['user']."'");
  371.  
  372.     // clear entries for this user from user_permissions table
  373.     $deleteQuery = "DELETE FROM user_permissions WHERE uid='".$uid."'";
  374.     commandQuery($deleteQuery);
  375.  
  376.     // insert current entries into user_permissions
  377.     $success = true;
  378.     foreach ($_POST['permissions'] as $permission) {
  379.         $insertQuery = "INSERT INTO user_permissions SET uid='$uid', permissionID='$permission'";
  380.         if ( !commandQuery($insertQuery) ) { $success = false; }
  381.     }
  382.     if ( $success === false ) {
  383.         $displayMessage .= '<p class="error">There was an error while trying to update the database</p>'."\n";
  384.     } else {
  385.         $displayMessage .= '<p class="confirmation">Database successfully updated</p>'."\n";
  386.     }
  387.  
  388.     // update .htpasswd files
  389.     if ( make_htpasswd() ) {
  390.         $displayMessage .= '<p class="confirmation">.htpasswd files successfully updated</p>'."\n";
  391.     } else {
  392.         $displayMessage .= '<p class="error">There was an error while trying to update the .htpasswd files</p>'."\n";
  393.     }
  394.  
  395.     // account status
  396.     if ( !commandQuery("UPDATE users SET accountStatus='".$_POST['status']."' WHERE uid='$uid'") ) {
  397.         $displayMessage .= '<p class="error">User status could not be updated</p>'."\n";

  402.     $_GET['step'] = '';

  404. }
