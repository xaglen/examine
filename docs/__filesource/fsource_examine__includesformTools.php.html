  11. /*****
  12.  * makeDropDown()
  13.  * Will generate correct html code for a select drop-down list
  14.  * from data stored in a database
  15.  * Inputs:    name - name of the field in the html form
  16.  *             DBtable - table where the needed data is stored
  17.  *             field - name of the field with value to display
  18.  *             idField - name of the field with value to store
  19.  *             default - text for first item on select list (doesn't have a value)
  20.  *             order - SQL for how to order the results
  21.  *             value - if the form was submitted but needs to be redisplayed (has value in $_POST) 
  22.  *****/
  23. function makeDropDown ( $name, $DBtable, $field, $idField, $default, $order, $value="" ) {
  24.     // query to get needed data
  25.     $query = "SELECT $idField, $field FROM $DBtable";
  26.     if ( $order ) $query .= " ".$order;
  27.     // get the information
  28.     $result = returnQuery($query);
  29.     if ( $result === FALSE ) {
  30.         echo '<p class="error">Error: could not access data for ' . $name . '.</p>'."\n";
  31.     } else {    // make drop-down
  32.         echo '<select name="' . $name . '">' . "\n";
  33.         echo "\t".'<option value="">' . $default . '</option>'."\n";
  34.         while ( $row = $result->fetchRow() ) {
  35.             if ( $value == $row[0] ) $selected = ' selected="selected"'; else $selected = '';
  36.             echo "\t" . '<option value="' . $row[0] . '"' . $selected . '>' . $row[1] . '</option>'."\n";
  37.         }
  38.         echo '</select>'."\n\n";
  39.     }
  40. }

  172. /***************
  173.  * errorMessages()
  174.  * Displays error messages for a section of the form
  175.  * Input:  accepts a variable-length list of arguments
  176.  ***************/
  177. function errorMessages() {
  178.     $numArgs = func_num_args();    // gets the number of arguments passed to the function
  179.     if ( $numArgs > 1 ) {    // more than one argument
  180.         $messages = func_get_args();
  181.         $hasError = false;
  182.         // check to see if any of the arguments did contain an error message
  183.         foreach ( $messages as $value ) if ( $value ) $hasError = true;
  184.         // if there was at least one error message to display, 
  185.         // output a new row with all of the error messages
  186.         if ( $hasError ) {
  187.             for ($i = 0; $i < $numArgs; $i++) {
  188.                 if ( $messages[$i] ) echo $messages[$i].'<br />'."\n";    // output a line if there is an error message
  189.             }
  190.         }
  191.     } elseif ( $numArgs == 1 ) {    // only one argument
  192.         $message = func_get_arg(0);
  193.         // if the argument contained an error message, output a row to display it
  194.         if ( $message ) {
  195.             echo $message;
  196.         }
  197.     }
  198.     
  199.     // for 0 arguments, function does nothing
  200. }

  211.  * pageNumberNav()
  212.  * Displays navigation links to multiple pages of results
  213.  *****/
  214. function pageNumberNav ( $totalPages, $currPage, $url='' ) {
  215.     $content = '<div class="pageNav">'."\n";
  216.  
  217.     // display link for FIRST page
  218.     if ( $currPage > 1 ) {
  219.         $link = $currPage - 1;
  220.         $content .= '<a href="?'.$url.'page='.$link.'" class="prev">'
  221.             . '&lt; prev</a>'."\n";
  222.     } else {
  223.         $content .= '<span class="prev">&lt; prev</span>'."\n";
  224.     }
  225.  
  226.     // display current page number and 
  227.     for ( $i=$currPage-2; $i<=$currPage+2; $i++ ) {
  228.         if ( $i > 0 && $i<=$totalPages ) {
  229.             if ( $currPage == $i ) {
  230.                 $content .= '<span class="thisPage">'.$i.'</span>'."\n";
  231.             } else {
  232.                 $content .= '<a href="?'.$url.'page='.$i.'">'.$i.'</a>'."\n";
  233.             }
  234.         }
  235.     }
  236.  
  237.     // display link for LAST page
  238.     if ( $currPage < $totalPages ) {
  239.         $link = $currPage + 1;
  240.         $content .= '<a href="?'.$url.'page='.$link.'" class="next">'
  241.             . 'next &gt;</a>'."\n";
  242.     } else {
  243.         $content .= '<span class="next">next &gt;</span>'."\n";
  244.     }
  245.  
  246.     $content .= '</div>'."\n\n";
  247.  
  248.     return $content;
  249. }
