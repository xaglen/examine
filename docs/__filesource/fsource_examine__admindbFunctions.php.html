  107.  * Note:  to decrypt a string, the iv must be saved when

  110. function encryptString ( $data, $iv='' ) {
  111.     // Open the cipher
  112.     $td = mcrypt_module_open('rijndael-128', '', 'ecb', '');
  113.  
  114.     // Create the IV and determine the keysize length
  115.     if ( $iv == '' || $iv == null ) {
  116.         srand();
  117.         $iv = mcrypt_create_iv(mcrypt_enc_get_iv_size($td), MCRYPT_RAND);
  118.     }
  119.     $ks = mcrypt_enc_get_key_size($td);
  120.  
  121.     // Create key
  122.     $key = substr(md5('chi alphA secrets'), 0, $ks);
  123.  
  124.     // Intialize encryption
  125.     mcrypt_generic_init($td, $key, $iv);
  126.  
  127.     // Encrypt data
  128.     $encrypted = mcrypt_generic($td, $data);
  129.  
  130.     // Terminate encryption handler
  131.     mcrypt_generic_deinit($td);
  132.  
  133.     return array($encrypted, $iv);
